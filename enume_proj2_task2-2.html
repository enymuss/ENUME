function [newRow] = zeroFristElement (row1, row2)
rowMultiplier = row2(1,1) / row1(1,1);
newRow = row2' - rowMultiplier * row1';
newRow = newRow';
end
function [newSum] = sumPreviousElements (colA, colX)
newVec = colA .* colX;
newSum = sum(newVec, 'all');
end
